CREATE TABLE Locador (
  idLocador INT PRIMARY KEY IDENTITY,
  nomeLocador VARCHAR(50) NOT NULL,
  cpfLocador CHAR(11) NOT NULL,
  emailLocador VARCHAR(60),
  telefoneLocador VARCHAR(15),
  senhaLocador VARCHAR(50) NOT NULL
);

CREATE TABLE Requerimento (
  idRequisito INT PRIMARY KEY IDENTITY,
  nomeRequisito VARCHAR(60) NOT NULL,
  descRequisito TEXT,
  statusRequerimento VARCHAR(20),
  idLocador INT NOT NULL,
  nomeLocador VARCHAR(50),
  FOREIGN KEY (idLocador) REFERENCES Locador(idLocador)
);
